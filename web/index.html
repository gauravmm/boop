function boop {
        local title=$(echo -ne "$1" | xxd -plain | tr -d '\n' | sed 's/\(..\)/%\1/g')
        local text=$(echo -ne "$2" | xxd -plain | tr -d '\n' | sed 's/\(..\)/%\1/g')
        local secret=$(cat ~/.boop)
        local sig=$(echo -ne "$secret:$title:$text" | sha224sum | cut -d" " -f1)
        wget -qO - "/push//$sig/$title/$text/$args" 1>/dev/null
}
